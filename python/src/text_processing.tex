\section{Text processing}
\subsection{Regular expressions (\python{re})}
\begin{compactitem}
	\item \python{search(pattern, string)} scans through a string, looking for any
place where the regex matches.
	\item \python{match(pattern, string)} checks whether the regex matches at the
beginning of the string.
	\item \python{findall(pattern, string)} returns all substrings where the regex
matches as a list
	\item \python{finditer(pattern, string)} returns all substrings where the regex
matches as an iterator.
\end{compactitem}
