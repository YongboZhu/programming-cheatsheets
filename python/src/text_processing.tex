\subsection{Regular expressions (\mintinline{python}{re})}
\begin{compactitem}
	\item \mintinline{python}{search(pattern, string)} scans through a string, looking for any
place where the regex matches.
	\item \mintinline{python}{match(pattern, string)} checks whether the regex matches at the
beginning of the string.
	\item \mintinline{python}{findall(pattern, string)} returns all substrings where the regex
matches as a list
	\item \mintinline{python}{finditer(pattern, string)} returns all substrings where the regex
matches as an iterator.
\end{compactitem}
