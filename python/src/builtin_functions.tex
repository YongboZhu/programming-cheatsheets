\section{Built-in functions}
\begin{compactitem}
	\item \python{all}/\python{any(iterable)}: ``$\forall$/$\exists$''.
	\item \python{input(prompt)}.
	\item \python{len(seq)}: the number of items of an object.
	\item \python{min}/\python{max(iterable)}: the smallest/largest item.
	\item \python{print(text, sep = " ", end = "{\textbackslash}n", file=sys.stdout)}
	\item \python{sorted(iterable)}: a new sorted list.
	\item \python{zip(*iterables)}: iterator of tuples.
	\item \python{reversed(seq)}.
\end{compactitem}
